```java
import java.util.*;

public class Programming {
	
	static int N;
	static int[][] arr;
	static int[][] dp;
	
	static int min = Integer.MAX_VALUE;
	
	public static void main(String[] args) {

		Scanner sc = new Scanner(System.in);
		
		N = sc.nextInt();
		dp = new int[N][3];
		arr = new int[N][3];
		
		for(int i=0; i<N; i++) {
			for(int j=0; j<3; j++) {
				arr[i][j] = sc.nextInt();
			}
		}
		//첫번째줄 RGB비용 초기화
		for(int i=0; i<3; i++) {
			dp[0][i] = arr[0][i];
		}
		
		for(int i=1; i<N; i++) {
			dp[i][0] = arr[i][0] + Math.min(dp[i-1][1], dp[i-1][2]);
			dp[i][1] = arr[i][1] + Math.min(dp[i-1][0], dp[i-1][2]);
			dp[i][2] = arr[i][2] + Math.min(dp[i-1][0], dp[i-1][1]);
		}
		
		System.out.println(Math.min(dp[N-1][0], Math.min(dp[N-1][1], dp[N-1][2])));
	}
}
```

//DFS 시간초과


```java


import java.util.*;

public class Main {
	
	static int N;
	static int[][] rgb;
	
	static int min = Integer.MAX_VALUE;
	
	public static void main(String[] args) {

		Scanner sc = new Scanner(System.in);
		
		N = sc.nextInt();
		rgb = new int[N][3];
		
		for(int i=0; i<N; i++) {
			for(int j=0; j<3; j++) {
				rgb[i][j] = sc.nextInt();
			}
		}
		
		for(int i=0; i<3; i++) {
			go(i, rgb[0][i], 1);
		}
		
		System.out.println(min);
	}
	
	static void go(int index, int sum, int start) {
		
		if(start==N) {
			min = Math.min(sum, min);
			return;
		}
		
		for(int i=0; i<3; i++) {

			if(index==i) continue;
			go(i, sum+rgb[start][i], start+1);
		}
	}
}
```
