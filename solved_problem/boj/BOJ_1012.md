import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.Scanner;

public class Main {	
	
	static int N;
	static int bugCount = 0;
	static int[][] map;
	static boolean[][] visit;
	
	static int[] dx = {0,-1,0,1};
	static int[] dy = {1,0,-1,0};
	
	public static void main(String[] args) throws NumberFormatException, IOException {

		Scanner sc = new Scanner(System.in);
		
		N = sc.nextInt();
		int col = 0;
		int row = 0;
		int count = 0;
		
		for(int i=0; i<N; i++) {
			
			//입력
			col = sc.nextInt();
			row = sc.nextInt();
			count = sc.nextInt();

			//map만들기
			
			map = new int[col][row];
			visit = new boolean[col][row];
			setupMap(map, col, row);
			
			for(int j=0; j<count; j++) {
				int x = sc.nextInt();
				int y = sc.nextInt();
				map[x][y] = 1;
			}
			
			for(int x=0; x<col; x++) {
				for(int y=0; y<row; y++) {
					if(map[x][y]==1 && !visit[x][y]) {
						search(col,row,x,y);
					}
				}
			}
			
			System.out.println(bugCount);
			
			//지렁이 수 출력 후 초기화
			bugCount = 0;
			
			//input 잘 들어가는지 확인
			//print(map, col, row);
		}
	}
	
	static void setupMap(int[][] map, int col, int row) {
		for(int i=0; i<col; i++) {
			for(int j=0; j<row; j++) {
				map[i][j] = 0;
			}
		}
	}
	
	static void print(int[][] arr, int col, int row) {
		for(int i=0; i<col; i++) {
			for(int j=0; j<row; j++) {
				System.out.print(arr[i][j]);
			}
			System.out.println();
		}
	}
	
	static void search(int col, int row, int i, int j) {
		
		Queue<Point> queue = new LinkedList<Point>();
		
		queue.add(new Point(i,j));
		visit[i][j] = true;
		
		while(!queue.isEmpty()) {
			
			Point now = queue.poll();
			
			for(int pos=0; pos<4; pos++) {
				
				int nextX = now.x + dx[pos];
				int nextY = now.y + dy[pos];
				
				if(nextX<0 || nextY<0 || nextX>=col || nextY>=row) {
					continue;
				}
				
				if(visit[nextX][nextY] || map[nextX][nextY]==0) {
					continue;
				}
				
				queue.add(new Point(nextX, nextY));
				visit[nextX][nextY] = true;
			}
		}
		
		bugCount++;
	}
}

class Point {
	int x;
	int y;
	
	Point(int x, int y) {
		this.x = x;
		this.y = y;
	}
}
