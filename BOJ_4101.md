import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {
	
	static int N, M;
	
	static String[] chess = {"BWBWBWBW","WBWBWBWB"};
	static char Map[][];
	
	public static void main(String[] args) throws IOException {

		//https://www.acmicpc.net/problem/4101
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

		String[] NM = br.readLine().split(" ");
		N = Integer.valueOf(NM[0]);
		M = Integer.valueOf(NM[1]);
		
		Map = new char[N][M];
		
		for(int i=0; i<N; i++) {
			String[] line = br.readLine().split("");
			for(int j=0; j<M; j++) {
				Map[i][j] = line[j].charAt(0);
			}
		}

		//8x8 정사각형을 0~N, 0~M 범위안에서 다 돌려보기
		search();
	}
	
	static void search() {
		
		int min = Integer.MAX_VALUE;
		
		for(int i=0; i<=N-8; i++) {
			for(int j=0; j<=M-8; j++) {
				for(int k=0; k<=1; k++) {
					min = Math.min(min, check(i,j,k));
				}
			}
		}
		
		if(min==Integer.MAX_VALUE) {
			System.out.println(0);
		} else {
			System.out.println(min);
		}
	}
	
	static int check(int x, int y, int index) {
		
		int count = 0;
		String compare = "";
		
		for(int i=x; i<x+8; i++) {
			compare = getChessLine(index+i);
			int strIndex = 0;
			for(int j=y; j<y+8; j++) {
				if(Map[i][j]!=compare.charAt(strIndex++)) {
					count++;
				}
			}
		}
		
		return count++;
	}
	
	static String getChessLine(int index) {
		if(index%2==0) {
			return chess[0];
		} else {
			return chess[1];
		}
	}
	
	static void print() {
		for(int i=0; i<N; i++) {
			for(int j=0; j<M; j++) {
				System.out.print(Map[i][j]+" ");
			}
			System.out.println();
		}
	}
}
